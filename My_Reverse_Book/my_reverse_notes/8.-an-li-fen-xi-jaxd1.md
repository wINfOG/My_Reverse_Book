```java
public class test {
    public static void main(String[] args) {
        Scanner myInput = new Scanner( System.in );

        while (y > 0) {
            while (a > 50) {
                a = a + 1;

                    if (a > 100) {
                        a = a * 5;
                        break;

                }
            }

        }
        System.out.println(y);

}
```

```
public static void main(String[] strArr) {
    int nextInt = new Scanner(System.in).nextInt();
    int i = 1;
    while (i > 0) {
        while (true) {
            if (nextInt > 50) {
                nextInt++;
                i++;
                if (nextInt > 10) {
                    if (nextInt > 100) {
                        nextInt *= 5;
                        break;
                    }
                    i /= nextInt;
                }
            }
        }
    }
    System.out.println(i);
}
```
